JavaScript / Node.js

const res = await fetch('https://api.opencurrencyx.com/v1/convert', {
  method: 'GET',
  params: { from: 'USD', to: 'EUR', amount: 100 }
});
const data = await res.json();
console.log(data.result);

Python

import requests

response = requests.get(
    'https://api.opencurrencyx.com/v1/convert', 
    params={'from': 'USD', 'to': 'EUR', 'amount': 100}
)
print(response.json()['result'])
